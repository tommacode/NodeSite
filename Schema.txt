Logs(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    IP VARCHAR(64),
    ForwardedFor VARCHAR(64),
    UserAgent VARCHAR(255),
    Method VARCHAR(8),
    Path VARCHAR(255),
    StatusCode INT
);

Projects(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    Title VARCHAR(128),
    Appetizer VARCHAR(255),
    Content TEXT,
    Tags VARCHAR(255),
    Comments TEXT,
    Likes INT,
    Status VARCHAR(8)
);

Comments(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    Project VARCHAR(128),
    Name VARCHAR(16),
    Email VARCHAR(255),
    Content TEXT,
    Likes INT
);

Users(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(16),
    Email VARCHAR(255),
    Password VARCHAR(255),
    Status VARCHAR(8),
    Sudo BOOLEAN
);

Sessions(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    User INT,
    Token VARCHAR(255),
    IP VARCHAR(64),
    UserAgent VARCHAR(255)
);
