create table Logs(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    IP VARCHAR(64),
    ForwardedFor VARCHAR(64),
    UserAgent VARCHAR(255),
    Method VARCHAR(8),
    Path VARCHAR(255),
    StatusCode INT
);

create table Projects(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    Title VARCHAR(128),
    Appetizer VARCHAR(255),
    Content TEXT,
    Tags VARCHAR(255),
    Comments TEXT,
    Likes INT,
    Status VARCHAR(8)
);

create table Comments(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    Project VARCHAR(128),
    UserID INT,
    Content TEXT,
    Likes INT,
    unique_id VARCHAR(255) NOT NULL
);

create table Users(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    Username VARCHAR(16),
    Email VARCHAR(255),
    Password VARCHAR(255),
    Sudo INT
);

create table Sessions(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Cookie VARCHAR(255),
    UserID INT,
    Time TIMESTAMP
);

create table projectLikes(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Project VARCHAR(128),
    UserID INT
);

create table commentLikes(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Unique_id VARCHAR(255),
    UserID INT
);
