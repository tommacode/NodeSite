create table Logs(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    IP VARCHAR(64),
    ForwardedFor VARCHAR(64),
    UserAgent VARCHAR(255),
    Method VARCHAR(8),
    Path VARCHAR(255),
    StatusCode INT,
    User INT
);

create table Projects(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    Title VARCHAR(128),
    Appetizer VARCHAR(255),
    Content TEXT,
    Tags VARCHAR(255),
    Comments TEXT,
    Likes INT,
    Status VARCHAR(8)
);

create table CommentsTest(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP,
    Project INT,
    UserID INT,
    Content TEXT,
    Likes INT,
    unique_id VARCHAR(255) NOT NULL
);

create table Users(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    Username VARCHAR(16),
    Email VARCHAR(255),
    Password VARCHAR(255),
    Sudo INT,
    ProfilePicture VARCHAR(255)
    ModifyProfilePicture INT,
    WriteComments INT,
    Locked INT,
);

create table Sessions(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Cookie VARCHAR(255),
    UserID INT,
    TimeCreated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    TimeLastUsed TIMESTAMP,
    IPCreatedWith VARCHAR(64),
    UserAgentCreatedWith VARCHAR(255),
    IPLastSeen VARCHAR(64),
    UserAgentLastSeen VARCHAR(255)
);

create table projectLikes(
    Project INT,
    UserID INT
);

create table commentLikes(
    Unique_id VARCHAR(255),
    UserID INT
);
